class Solution {
public:
    vector<int> runningSum(vector<int>& nums) {
    vector<int> result;
    vector<int>::iterator ptr = nums.begin();
    vector<int>::iterator ftr = nums.end();
    int cpt=0;
    while (ptr != ftr){
        cpt+=*ptr;
        result.push_back(cpt);
        advance(ptr,1);
    }
    return result;

}
    
};
